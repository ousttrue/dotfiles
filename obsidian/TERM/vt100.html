C¶

isatty Linux

#include <stdio.h>
#include <unistd.h>
int main(int argc, char *argv[]) { 
	if (isatty(fileno(stdout))) { 
		fputs("Detected a TTY\n", stderr);
		fputs("\033[31mRED \033[32mGREEN \033[34mBLUE\033[0m\n", stdout); 
	} else { 
		fputs("Not a TTY\n", stderr); 
	} 
}
