janus https¶

${PREFIX}/etc/janus/janus.transport.http.jcfg

general: {
        https = true
        secure_port = 8089
}
certificates: {
        cert_pem = "/etc/ssl/certs/ssl-cert-snakeoil.pem"
        cert_key = "/etc/ssl/private/ssl-cert-snakeoil.key"
}
