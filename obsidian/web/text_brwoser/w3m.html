size_t wtf_len(wc_uchar *p)
{
    wc_uchar *q = p;
    wc_uchar *strz = p + strlen((const char*)p);

    q += WTF_LEN_MAP[*q];
    while (q < strz && ! WTF_WIDTH_MAP[*q])
	q += WTF_LEN_MAP[*q];
    return q - p;
}
