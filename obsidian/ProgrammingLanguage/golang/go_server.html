import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [react()],
  server: {
    proxy: {
      '/ws': {
        target: 'ws://localhost:8080',
        ws: true,
      },
      "/api": {
        target: "http://localhost:8080", // <- to golang backend
        changeOrigin: true,
      },    }
  }
})
