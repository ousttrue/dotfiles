		auto result = LeapPollConnection(connectionHandle, timeout, &msg);
		//Handle message
	}
}

## LEAP_HAND
```c
enum eLeapHandType {
  eLeapHandType_Left,
  eLeapHandType_Right
};

struct LEAP_HAND
{
	uint32_t id;
	eLeapHandType type;

	LEAP_DIGIT thumb;
	LEAP_DIGIT index;
	LEAP_DIGIT middle;
	LEAP_DIGIT ring;
	LEAP_DIGIT pinky;
	LEAP_BONE arm;
	LEAP_PALM palm;
};
```

struct LEAP_BONE
{
	LEAP_VECTOR prev_joint;
	LEAP_VECTOR next_joint;
	float width;
	LEAP_QUATERNION rotation;

{
	float x;
	float y;
	float z;
};
