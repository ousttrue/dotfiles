## terms.cpp

```c
void addnstr(char *s, int n) {
  int i;
  int len, width;

  for (i = 0; *s != '\0';) {
    width = wtf_width((wc_uchar *)s);
    if (i + width > n)
      break;
    len = wtf_len((wc_uchar *)s); // <- これ
    addmch(s, len);
    s += len;
    i += width;
  }
}
```
## libwc/wtf.cpp

```cpp
size_t wtf_len(wc_uchar *p)
{
    wc_uchar *q = p;
    wc_uchar *strz = p + strlen((const char*)p);

    q += WTF_LEN_MAP[*q];
    while (q < strz && ! WTF_WIDTH_MAP[*q])
	q += WTF_LEN_MAP[*q];
    return q - p;
}
```
