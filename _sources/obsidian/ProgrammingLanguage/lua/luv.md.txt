## iter dir
```lua
  local fs, err, msg = uv.fs_scandir(path)
  if err then
    print(err, msg)
    return
  end
  if fs then
    while true do
      local name, type, err, msg = uv.fs_scandir_next(fs)
      if not name then
        break
      end
      print(name, type)
    end
  end
```
